Declarar varios tipos de IVA

const Afip = require('@afipsdk/afip.js');

const afip = new Afip({ 'CUIT': 11111111111 });

(async () => {

	/**
	 * Número del ultimo comprobante
	 **/
	const last_voucher = await afip.ElectronicBilling.getLastVoucher(1,6);

	const data = {
		'CantReg' 	: 1,  // Cantidad de comprobantes a registrar
		'PtoVta' 	: 1,  // Punto de venta
		'CbteTipo' 	: 6,  // Tipo de comprobante  
		'Concepto' 	: 1,  // Concepto del Comprobante: (1)Productos, (2)Servicios, (3)Productos y Servicios
		'DocTipo' 	: 99, // Tipo de documento del comprador (99 consumidor final)
		'DocNro' 	: 0,  // Número de documento del comprador (0 consumidor final)
		'CbteDesde' : last_voucher+1,  // Número de comprobante
		'CbteHasta' : last_voucher+1,  // Número de comprobante
		'CbteFch' 	: parseInt(new Date(Date.now() - ((new Date()).getTimezoneOffset() * 60000)).toISOString().split('T')[0].replace(/-/g, '')), // (Opcional) Fecha del comprobante (yyyymmdd) o fecha actual si es nulo
		'ImpTotal' 	: 231.5, // Importe total del comprobante
		'ImpTotConc': 0,   // Importe neto no gravado
		'ImpNeto' 	: 200, // Importe neto gravado
		'ImpOpEx' 	: 0,   // Importe exento de IVA
		'ImpIVA' 	: 31.5,  //Importe total de IVA
		'ImpTrib' 	: 0,   //Importe total de tributos
		'MonId' 	: 'PES', //Tipo de moneda usada en el comprobante ('PES' para pesos argentinos) 
		'MonCotiz' 	: 1,     // Cotización de la moneda usada (1 para pesos argentinos)  
		'Iva' 		: [ // (Opcional) Alícuotas asociadas al comprobante
			{
				'Id' 		: 5, // Id del tipo de IVA (5 para 21%) 
				'BaseImp' 	: 100, // Base imponible
				'Importe' 	: 21 // Importe 
			},
			{
				'Id' 		: 4, // Id del tipo de IVA (4 para 10.5%) 
				'BaseImp' 	: 100, // Base imponible
				'Importe' 	: 10.5 // Importe 
			}
		], 
	};

	/** 
	 * Creamos el comprobante 
	 **/
	const res = await afip.ElectronicBilling.createVoucher(data);

	/**
	 * Mostramos por pantalla los datos de la nueva Factura 
	 **/
	console.log({
		'cae' : res.CAE, //CAE asignado a la Factura
		'vencimiento' : res.CAEFchVto //Fecha de vencimiento del CAE
	});
})();

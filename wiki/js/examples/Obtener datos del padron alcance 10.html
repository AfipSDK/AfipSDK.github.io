Obtener datos del padrón alcance 10

const Afip = require('@afipsdk/afip.js');

const afip = new Afip({ 'CUIT': 11111111111 });

(async () => {
	/**
	 * CUIT a consultar
	 **/
	const cuit = 30500001735;

	/**
	 * Obtenemos los datos del contribuyente
	 **/
	const datos = await afip.RegisterScopeTen.getTaxpayerDetails(cuit);

	if(!datos){
		console.log('El contribuyente no existe en el padrón alcance 10.');
	}
	else{
		/**
		 * Mostramos por pantalla los datos del contribuyente
		 **/
		console.log(datos);
	}
})();
